function vcInitState(){ const words = vcGetVerseWords(); // Start with 2–3 words already “built” const startCount = Math.min(words.length, 2 + Math.floor(Math.random() * 2)); // 2 or 3 const built = words.slice(0, startCount); const nextIndex = startCount; const parts = vcParseVerseRefParts(); const state = { game: "verse_chain", // main verse build words, built, nextIndex, // end questions phase: "words", // "words" -> "book" -> "ref" -> done correctBook: parts.book, // e.g. "Luke" correctCv: parts.cv, // e.g. "2:11" // options carousel options: [], optionIndex: 0, // feedback done: false, lastPickCorrect: null, // verse-stage reserved height tracking (used by vcEnsureReservedVerseHeight) _vcStageWidth: 0, _vcStageHeight: 0 }; vcMakeNewOptions(state); return state; }
